/**
 * https://leetcode.com/problems/contains-duplicate/
 * @param {number[]} nums
 * @return {boolean}
 */
const containsDuplicate = (nums, numsSet = new Set()) => {
  for (const num of nums) {
    if (numsSet.has(num)) return true;

    numsSet.add(num);
  }

  return false;
};
